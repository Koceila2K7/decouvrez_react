Dans le dossier pages créez le fichier Todo.jsx

collez le code suivant :

/client/pages/Todo.jsx
import { Box, Button, Container, Typography } from "@mui/material";
import React, { useEffect, useState } from "react";
import DeleteIcon from "@mui/icons-material/Delete";
import { useParams } from "react-router-dom";
import { getTodoById } from "../api/todo.api";
export default function Todo() {
  const { id } = useParams();
  const [error, setError] = useState(false);
  const fetchTodoById = async () => {
    try {
      const { data } = await getTodoById(id);
      setTodo(data);
      setError(false);
    } catch (e) {
      console.log(e);
      setError(true);
    }
  };

  useEffect(() => {
    fetchTodoById();
  }, []);

  const [todo, setTodo] = useState({
    titre: "",
    desc: "",
    couleur: "",
  });

  return (
    <Box>
      {!error ? (
        <Container>
          <Typography variant="h4">{todo.titre}</Typography>

          <Typography variant="body1">{todo.desc}</Typography>
          <div
            style={{
              width: 50,
              height: 50,
              background: todo.couleur,
              borderRadius: 5,
              marginBottom: 10,
            }}
          ></div>
          <Button variant="contained" endIcon={<DeleteIcon />}>
            Supprimer
          </Button>
        </Container>
      ) : (
        <Container>
          <Typography variant="h4">Erreur de récupération</Typography>
        </Container>
      )}
    </Box>
  );
}

Remplacez le code du fichier App.js par le code suivant :

App.js
import React, { useState } from "react";
import AppNavBar from "./components/AppNavBar";

import { Box } from "@mui/material";
import { ThemeProvider } from "@emotion/react";
import { createTheme } from "@mui/material";
import { deepOrange } from "@mui/material/colors";
import Main from "./pages/Main";
import useAuth from "./hooks/useAuth";
import { BrowserRouter as Router, Routes, Route } from "react-router-dom";
import Todo from "./pages/Todo";

const themeLight = createTheme({
  palette: {
    primary: { main: deepOrange[900] },
  },
});
const themeDark = createTheme({
  palette: {
    primary: { main: deepOrange[900] },
    mode: "dark",
  },
});
export default function App() {
  const [isDark, setIsDark] = useState(false);
  const switchTheme = () => setIsDark((old) => !old);
  const [isLoged, logout] = useAuth();
  return (
    <ThemeProvider theme={isDark ? themeDark : themeLight}>
      {isLoged ? (
        <Box
          sx={{
            bgcolor: "background.paper",
            height: "100vh",
          }}
        >
          <AppNavBar handleDark={switchTheme} handleLogout={logout} />
          <Router>
            <Routes>
              <Route element={<Todo/>} path="/todo/:id"/>
              <Route element={<Main />} path="*" />
            </Routes>
          </Router>
        </Box>
      ) : (
        <h1>non connecté</h1>
      )}
    </ThemeProvider>
  );
}

Remplacez le code du fichier Main.jsx par le code suivant pour ajouter le code de navigation suite au clique sur une todo.

Main.jsx
import React, { useEffect, useState } from "react";
import AddBtn from "../components/AddBtn";
import TodoCard from "../components/TodoCard";
import { Box, Container, Grid } from "@mui/material";
import FormAdd from "../components/FormAdd";
import { getTodos } from "../api/todo.api";
import { useNavigate } from "react-router-dom";

export default function Main() {
  const [formVisible, setFormVisible] = useState(false);
  const [todos, setTodos] = useState([]);
  const fetchTodos = async () => {
    const { data } = await getTodos();
    setTodos(data);
  };
  const navigate = useNavigate();
  const navigateTo = (id) => navigate(`/todo/${id}`);
  useEffect(() => {
    fetchTodos();
  }, []);

  const handleFormOpen = () => setFormVisible(true);
  return (
    <Box
      sx={{
        bgcolor: "background.paper",
        height: "100vh",
      }}
    >
      <Container>
        <Grid container>
          {todos.map((e) => (
            <TodoCard key={e._id} {...e} id={e._id} onClick={navigateTo} />
          ))}
        </Grid>
      </Container>
      <FormAdd open={formVisible} setOpen={setFormVisible} />
      <AddBtn onClick={handleFormOpen} />
    </Box>
  );
}
