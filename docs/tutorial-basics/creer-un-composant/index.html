Créez le squelette

Création du squelette de l'application

Bonnes pratiques

ReactJS n'impose pas une façon de gérer les composants et les fichiers. Il est vivement recommandé de mettre les composants dans un dossier "components" et les vues dans un dossier "pages".

Création des composants​

Nav bar :​

Créez le fichier AppNavBar.jsx dans le dossier components :

/client/src/components
touh AppNavBar.jsx

Collez le code suivant :

/client/src/components/AppNavBar.jsx
import * as React from "react";
import AppBar from "@mui/material/AppBar";
import Box from "@mui/material/Box";
import Toolbar from "@mui/material/Toolbar";
import Typography from "@mui/material/Typography";
import Button from "@mui/material/Button";

export default function AppNavBar({ handleLogout, handleDark }) {
  return (
    <Box sx={{ flexGrow: 1 }}>
      <AppBar position="static">
        <Toolbar>
          <Typography variant="h6" component="div" sx={{ flexGrow: 1 }}>
            Todos
          </Typography>
          <Button color="inherit" onClick={handleDark}>
            Dark
          </Button>
          <Button color="inherit" onClick={handleLogout}>
            Logout
          </Button>
        </Toolbar>
      </AppBar>
    </Box>
  );
}

Ce code est un extrait de la documentation-MUI, avec quelques petites modifications.

      </Card>

  );
}

Formulaire de création de Todo :​

Nous allons créer un formulaire sous forme d'une fenêtre modale. Créez le fichier FormAdd.jsx dans le dossier components :

/client/src/components
touh FormAdd.jsx

Collez le code suivant :

/client/src/components/FormAdd.jsx
import * as React from "react";
import Box from "@mui/material/Box";
import Typography from "@mui/material/Typography";
import Modal from "@mui/material/Modal";
import { Button, TextField } from "@mui/material";
import { useState } from "react";

const style = {
  position: "absolute",
  top: "50%",
  left: "50%",
  transform: "translate(-50%, -50%)",
  width: 400,
  bgcolor: "background.paper",
  boxShadow: 24,
  p: 4,
};

export default function FormAdd({ open, setOpen }) {
  const handleClose = () => setOpen(false);
  const [titre, setTitre] = useState("");
  const [desc, setDesc] = useState("");
  const [selectedColor, setSelectedColor] = useState("");
  const handleTitreChange = (e) => setTitre(e.target.value);
  const handleDescChange = (e) => setDesc(e.target.value);

  return (
    <div>
      <Modal
        open={open}
        onClose={handleClose}
        aria-labelledby="modal-modal-title"
        aria-describedby="modal-modal-description"
      >
        <Box sx={style}>
          <Typography variant="h5" style={{ marginBottom: 10 }}>
            Ajouter une todo :
          </Typography>
          <TextField
            value={titre}
            fullWidth
            label="Titre :"
            variant="outlined"
            style={{ marginBottom: 10 }}
            onChange={handleTitreChange}
          />
          <TextField
            value={desc}
            onChange={handleDescChange}
            fullWidth
            label="Desc :"
            variant="outlined"
            multiline
            minRows={2}
            style={{ marginBottom: 10 }}
          />
          <input
            type="color"
            style={{ width: "100%", display: "block",marginBottom: 15 }}
            value={selectedColor}
            onChange={(e) => setSelectedColor(e.target.value)}
          />
          <Button variant="contained" fullWidth>
            Ajouter
          </Button>
        </Box>
      </Modal>
    </div>
  );
}

Visualisez vos composants :​

Remplacez le code du fichier App.js par le code suivant :

/client/src/App.js

import React, { useState } from "react";
import AppNavBar from "./components/AppNavBar";
import AddBtn from "./components/AddBtn";
import TodoCard from "./components/TodoCard";
import { Container, Grid } from "@mui/material";
import FormAdd from "./components/FormAdd";
const todos = [
  {
    id: "1",
    titre: "Ma Todo",
    desc: "Super todo",
    couleur: "#F00",
  },

  {
    id: "2",
    titre: "Ma Todo 1",
    desc: "Super todo 2",
    couleur: "#0F0",
  },
  {
    id: "3",
    titre: "Ma Todo 1",
    desc: "Super todo 2",
    couleur: "#0F0",
  },
  {
    id: "4",
    titre: "Ma Todo 1",
    desc: "Super todo 2",
    couleur: "#0F0",
  },
];
export default function App() {
  const [formVisible, setFormVisible] = useState(false);
  const handleFormOpen = () => setFormVisible(true);
  return (
    <div>
      <AppNavBar />
      <Container>
        <Grid container>
          {todos.map((e) => (
            <TodoCard key={e.id} {...e} />
          ))}
        </Grid>
      </Container>
      <FormAdd open={formVisible} setOpen={setFormVisible} />
      <AddBtn onClick={handleFormOpen} />
    </div>
  );
}
