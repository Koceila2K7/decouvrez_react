Gérez vos appels réseau avec AXIOS

Maintenant nous allons connecter notre application au serveur d'api Rest

Bonnes pratiques :

Il est vivement recommandé de regrouper toute la logique de l'API dans un dossier "api".

Créez un dossier api dans src

/client/src
mkdri api

Dans le dossier api créez les deux fichiers index.js et todo.api.js

/client/src/api
touch index.js todo.api.js

Dans le fichier index.js collez le code suivant :

/client/src/api/index.js
export * from "./todo.api.js";

Dans le fichier todo.api.ts collez le code suivant :

/client/src/api/index.js
import axios from "axios";

const todoApiClient = axios.create({
  baseURL: "http://localhost:3333",
});

todoApiClient.interceptors.request.use((req) => {
  const token = sessionStorage.getItem("TOKEN");
  console.log({ token });
  req.headers.Authorization = `Bearer ${token}`;
  return req;
});

export const getTodos = () => {
  return todoApiClient.get("/todos");
};

export const createTodo = ({ titre, dec, couleur }) => {
  return todoApiClient.post("/todos", { titre, dec, couleur });
};

export const getTodoById = (id) => todoApiClient.get(`/todos/${id}`);
export const deleteTodoById = (id) => todoApiClient.delete(`/todos/${id}`);

Remplacez le code du fichier FormAdd.jsx par le code suivant :

/client/src/components/FormAdd.jsx
import * as React from "react";
import Box from "@mui/material/Box";
import Typography from "@mui/material/Typography";
import Modal from "@mui/material/Modal";
import { Button, TextField } from "@mui/material";
import { useState } from "react";
import { createTodo } from "../api/todo.api";

const style = {
  position: "absolute",
  top: "50%",
  left: "50%",
  transform: "translate(-50%, -50%)",
  width: 400,
  bgcolor: "background.paper",
  boxShadow: 24,
  p: 4,
};

export default function FormAdd({ open, setOpen }) {
  const handleClose = () => setOpen(false);
  const [titre, setTitre] = useState("");
  const [desc, setDesc] = useState("");
  const [selectedColor, setSelectedColor] = useState("#000");
  const handleTitreChange = (e) => setTitre(e.target.value);
  const handleDescChange = (e) => setDesc(e.target.value);

  const handleAjouter = async (e) => {
    try {
      await createTodo({ titre, desc, couleur: selectedColor });
      //Création avec succès
    } catch (error) {
      //echec de la création
    }
  };

  return (
    <div>
      <Modal
        open={open}
        onClose={handleClose}
        aria-labelledby="modal-modal-title"
        aria-describedby="modal-modal-description"
      >
        <Box sx={style}>
          <Typography variant="h5" style={{ marginBottom: 10 }}>
            Ajouter une todo :
          </Typography>
          <TextField
            value={titre}
            fullWidth
            label="Titre :"
            variant="outlined"
            style={{ marginBottom: 10 }}
            onChange={handleTitreChange}
          />
          <TextField
            value={desc}
            onChange={handleDescChange}
            fullWidth
            label="Desc :"
            variant="outlined"
            multiline
            minRows={2}
            style={{ marginBottom: 10 }}
          />
          <input
            type="color"
            style={{ width: "100%", display: "block", marginBottom: 15 }}
            value={selectedColor}
            onChange={(e) => setSelectedColor(e.target.value)}
          />
          <Button variant="contained" fullWidth onClick={handleAjouter}>
            Ajouter
          </Button>
        </Box>
      </Modal>
    </div>
  );
}
